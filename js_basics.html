📦 一、变量的定义 
// var：函数作用域，可重复声明
var city = "北京";
var city = "上海"; // 允许重复声明

// let：块作用域，不可重复声明
let score = 100;
score = 95;         // 可重新赋值

// const：常量，声明后不可重新赋值
const PI = 3.14159;

// 块作用域演示
{
  let blockVar = "块内变量";
  console.log("块内访问:", blockVar);
}
// console.log(blockVar); // 块外访问会报错!

console.log("city =", city);
console.log("score =", score);
console.log("PI =", PI);

🧮 三、表达式 
// 算术运算符
let a = 10, b = 3;
console.log(`${a}+${b}= ${a+b}, ${a}%${b}= ${a%b}, ${a}**${b}= ${a**b}`);

// 比较运算符 (== vs ===)
console.log("'5'== 5 :", '5' ==  5); // true  (宽松比较)
console.log("'5'=== 5:", '5' === 5); // false (严格比较)

// 逻辑运算符
console.log("true && false:", true && false);
console.log("null ?? 'default':", null ?? 'default');

// 三元表达式
let age = 20;
let status = age >= 18 ? "成年" : "未成年";
console.log("年龄状态:", status);

// 字符串模板
let name = "世界";
console.log(`你好，${name}！`);
🔀 四、流程控制 
// if / else if / else
let grade = 78;
if (grade >= 90)      console.log("优秀");
else if (grade >= 75) console.log("良好");
else                  console.log("待提高");

// switch
let day = new Date().getDay();
switch (day) {
  case 0: console.log("今天是周日"); break;
  case 6: console.log("今天是周六"); break;
  default: console.log(`今天是周${day}，工作日`);
}

// for 循环
let sum = 0;
for (let i = 1; i <= 5; i++) sum += i;
console.log("1~5 之和:", sum);

// while 循环
let n = 1, result = 1;
while (n <= 5) { result *= n++; }
console.log("5! =", result);

// for...of 遍历数组
const fruits = ["🍎苹果", "🍌香蕉", "🍊橙子"];
for (const f of fruits) console.log(f);
⚙️ 五、函数 
// 函数声明
function greet(name = "陌生人") {
  return `你好，${name}！`;
}

// 函数表达式
const square = function(x) { return x * x; };

// 箭头函数
const cube = x => x ** 3;

// 高阶函数（接收函数作为参数）
function applyTwice(fn, val) {
  return fn(fn(val));
}

// 闭包
function makeCounter() {
  let count = 0;
  return () => ++count;
}

// 运行示例
console.log(greet());
console.log(greet("JavaScript"));
console.log("square(7):", square(7));
console.log("cube(3):",   cube(3));
console.log("applyTwice(square,2):", applyTwice(square, 2));
const counter = makeCounter();
console.log("counter():", counter(), counter(), counter());
