① 变量的定义 var / let / const

var（函数作用域） let（块作用域） const（常量）
// var —— 函数作用域，可重复声明，存在变量提升 var city = "北京"; var city = "上海"; // 允许重复声明 // let —— 块作用域，不可重复声明 let score = 98; score = 100; // 可以重新赋值 // const —— 块作用域，声明后不可重新赋值 const PI = 3.14159; // PI = 3; → TypeError: Assignment to constant variable // 变量提升示例 console.log(hoisted); // undefined（不报错） var hoisted = "我被提升了";

③ 表达式 运算符

算术运算 比较运算 逻辑运算 三元运算 模板字符串
// 算术运算 let a = 10, b = 3; console.log(a + b, a - b, a * b, a / b, a % b, a ** b); // 比较运算（=== 严格相等，不做类型转换） console.log(5 == "5"); // true（宽松） console.log(5 === "5"); // false（严格） // 逻辑运算 console.log(true && false); // false console.log(true || false); // true console.log(!true); // false // 三元运算符 let age = 20; let status = age >= 18 ? "成年" : "未成年"; // 模板字符串 console.log(`年龄：${age}，状态：${status}`);

④ 流程控制 if / switch / for / while

// if / else if / else function grade(score) { if (score >= 90) return "优秀"; else if (score >= 75) return "良好"; else if (score >= 60) return "及格"; else return "不及格"; } // switch function dayName(n) { switch(n) { case 1: return "星期一"; case 2: return "星期二"; // ... default: return "未知"; } } // for 循环 / for...of / for...in const fruits = ["苹果", "香蕉", "橙子"]; for (let i = 0; i < fruits.length; i++) { /* ... */ } for (const f of fruits) { /* ... */ } // while / do...while let x = 0; while (x < 3) { x++; }

⑤ 函数 声明 / 表达式 / 箭头 / 闭包

// 1. 函数声明（存在提升） function add(a, b) { return a + b; } // 2. 函数表达式 const multiply = function(a, b) { return a * b; }; // 3. 箭头函数（简洁，不绑定 this） const square = n => n ** 2; const greet = (name, lang = "zh") => lang === "zh" ? `你好，${name}！` : `Hello, ${name}!`; // 4. 默认参数 & 剩余参数 const sum = (...nums) => nums.reduce((acc, v) => acc + v, 0); // 5. 闭包 —— 函数记住外部作用域的变量 function makeCounter() { let count = 0; return () => ++count; } const counter = makeCounter(); counter(); counter(); // 1, 2
